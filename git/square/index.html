const canvas = document.getElementById('canvas');
const ctx = canvas.getContext('2d');
canvas.width = 800;
canvas.height = 600;




const Square = function(x1,y1,x2,y2,color = 'black'){
//x1 начало рисунка по оси х,y1 начало рисунка по оси у
x2 конец рисунка по оси х,y2 конец рисунка по оси у,color цвет
ctx.fillStyle = color;
ctx.fillRect(x1, y1, x2, y2)
}
let Main = function() {
new Square(1,1,122,122,'black')
new Square(1,122,122,122,'red')
}
Main()
